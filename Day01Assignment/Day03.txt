How to compare two json have the same properties without order:
var obj1 = {
    name: "Person1",
    age: 5
};

var obj2 = {
    age: 5,
    name: "Person1"
}

function result(obj1, obj2) {
    var strObj1 = JSON.stringify(obj1);
    var strObj2 = JSON.stringify(obj2);
    return strObj1 === strObj2;
}
console.log(result(obj1, obj2));
